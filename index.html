Comprehensive Default Filtering.

AutoSort comes completely pre-configured with over 40+ of the most popular file extensions natively mapped to clean category folders. Customize these variables directly within the core config to tailor it exactly to your needs.

config.json
"categories": {
    "Images": [".jpg", ".jpeg", ".png", ".gif", ".webp", ".tiff", ".svg", ".ico"],
    "Documents": {
        "Word": [".docx", ".doc"],
        "PowerPoint": [".pptx", ".ppt", ".ppsx"],
        "PDF": [".pdf"],
        "Text": [".txt", ".odt", ".rtf"],
        "Excel": [".xls", ".xlsx", ".csv"]
    },
    "Archives": [".zip", ".rar", ".7z", ".tar", ".gz"],
    "Videos": [".mp4", ".mov", ".avi", ".mkv", ".webm", ".flv"],
    "Audio": [".mp3", ".wav", ".ogg", ".flac", ".aac"],
    "Setup": [".exe", ".msi", ".dmg"],
    "Code": [".py", ".html", ".css", ".js", ".php", ".json", ".java", ".cpp"]
}
